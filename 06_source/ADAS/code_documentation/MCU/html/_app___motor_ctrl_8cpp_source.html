130 void CMotorCtrl::StraightDrive(void)
131 {
132  //Checks if forward or backward state is active
133  if(rtObj.rtDW.is_c3_Chart == 2 || rtObj.rtDW.is_c3_Chart == 1){
134  //Set upper border
135  if(rtObj.rtDW.curr_angle + control_area >= 360){
136  upper_b = rtObj.rtDW.curr_angle + control_area - 360;
137  }else{
138  upper_b = rtObj.rtDW.curr_angle + control_area;
139  }
140  //Set lower border
141  if(rtObj.rtDW.curr_angle - control_area < 0){
142  lower_b = rtObj.rtDW.curr_angle - control_area + 360;
143  }else{
144  lower_b = rtObj.rtDW.curr_angle - control_area;
145  }
146  //Checks if the gyro signal is smaller than the initial angle and higher than the lower border (left drift)
147  if(rtObj.rtU.gyro_signal < rtObj.rtDW.curr_angle || rtObj.rtU.gyro_signal >= lower_b){
148  //Set a maximum difference between the adaption variables to prevent too high speed
149  if(adapt_l - adapt_r > -15 && adapt_r - adapt_l < 15 || ctrl_side == 1){
150  // Reset adaption variables at first enter of left drift adaption
151  if(ctrl_side == 1){
152  adapt_r = 0;
153  adapt_l = 0;
154  }
155  ctrl_side =2;
156  // Adapt the variables for forward state
157  if(rtObj.rtDW.is_c3_Chart == 2){
158  adapt_r=adapt_r+2;
159  adapt_l=adapt_l-2;
160  // Adapt the variables for backward state (reversed)
161  }else if(rtObj.rtDW.is_c3_Chart == 1){
162  adapt_r=adapt_r-2;
163  adapt_l=adapt_l+2;
164  }
165  }
166  //Checks if the gyro signal is bigge than the initial angle and smaller than the upper border (right drift)
167  }else if( rtObj.rtDW.curr_angle < rtObj.rtU.gyro_signal && rtObj.rtU.gyro_signal <= upper_b){
168  //Set a maximum difference between the adaption variables to prevent too high speed
169  if(adapt_l - adapt_r < 15 && adapt_r - adapt_l > -15 || ctrl_side == 2){

233 void CMotorCtrl::checkOverflow(void){
234  //Checks if the H-bridge is enables or the IDLE state is active
235  if (digitalRead(PIN_ENABLE) == HIGH || rtObj.rtDW.is_c3_Chart == 3) {
236  //Stop the motor if the condition is true
237  m_pwmUnitRight_o.writeDuty(LOW);
238  m_pwmUnitLeft_o.writeDuty(LOW);
239  output_l = output_r = 0;
240  } else {
241  //Checks if the right adaption is negative
242  if(adapt_r < 0){
243  //Checks if the absolute of the negative value is higher than the output which is written to the motor
244  if(abs(adapt_r) > output_r*2){
245  //Ignore adaption variables
246  m_pwmUnitRight_o.writeDuty(output_r*2);
247  m_pwmUnitLeft_o.writeDuty(output_l*2);
248  }else{
249  // Write to motor with adaption variables
250  m_pwmUnitRight_o.writeDuty(output_r*2+adapt_r);
251  m_pwmUnitLeft_o.writeDuty(output_l*2+adapt_l);
252  }
253  //Checks if the left adaption is negative
254  }else if(adapt_l < 0){
255  //Checks if the absolute of the negative value is higher than the output which is written to the motor
256  if(abs(adapt_l) > output_l*2){
257  //Ignore adaption variables

397 void CMotorCtrl::checkState(void) {
398  switch (rtObj.rtDW.is_c3_Chart) {
399  case 1:
400  // Case Backward state
401  // Set the direction pins of the H-bridge to drive backwards
402  digitalWrite(PIN_DIRECTION_L, LOW);
403  digitalWrite(PIN_DIRECTION_R, HIGH);
404  // Checks if first enter of this state is true and the speed is not set via the API function
405  if(curState != 1 && !speed_set){
406  // Clear the PI controller
407  myPID.clear();
408  // Reconfigure the PI controller
409  myPID.configure(KP_VALUE, KI_VALUE, KD_VALUE, HZ_VALUE, OUTPUT_BITS, OUTPUT_SIGNED);
410  // Set setpoint of the PI controller
411  setpoint_l = spd;
412  setpoint_r = spd;
413  }
414  curState = 1;
415  // Check if desired distance is reached
416  if(peak_sum_r >= d_way && peak_sum_l >= d_way){
417  // Adapt stateflow and CMotorCtrl parameters to perform a state change to IDLE
418  rtObj.rtU.dist=0;
419  d_way=0;
420  }
421  break;
422  case 2:
423  // Case Forward state
424  // Set the direction pins of the H-bridge to drive forward
425  digitalWrite(PIN_DIRECTION_L, HIGH);
426  digitalWrite(PIN_DIRECTION_R, LOW);
427  // Checks if first enter of this state is true and the speed is not set via the API function
428  if(curState != 2 && !speed_set){
429  // Clear the PI controller
430  myPID.clear();
431  // Reconfigure the PI controller
432  myPID.configure(KP_VALUE, KI_VALUE, KD_VALUE, HZ_VALUE, OUTPUT_BITS, OUTPUT_SIGNED);
433  // Set setpoint of the PI controller
434  setpoint_l = spd;
435  setpoint_r = spd;
436  }
437  curState = 2;
438  // Check if desired distance is reached
439  if(peak_sum_r >= d_way && peak_sum_l >= d_way){
440  // Adapt stateflow and CMotorCtrl parameters to perform a state change to IDLE
441  rtObj.rtU.dist=0;
442  d_way=0;
443  }
444  break;
445  case 3:
446  // Case IDLE state
447  // Set direction pins of H-bridge to high as default
448  digitalWrite(PIN_DIRECTION_L, HIGH);
449  digitalWrite(PIN_DIRECTION_R, HIGH);
450  // Stop motors
451  m_pwmUnitRight_o.writeDuty(LOW);
452  m_pwmUnitLeft_o.writeDuty(LOW);
453  // Checks if first enter of this state
454  if(curState != 3){
455  // Reset PI controller
456  myPID.clear();
457  // Set setpoint of PI controller to zero
458  setpoint_l = 0;
459  setpoint_r = 0;
460  }
461  // Checks if previous state was a turn state
462  if(curState == 4 || curState == 5){
463  DPRINT("Rotation reached!");
464  // Set turn parameter of stateflow to zero
465  rtObj.rtU.turn = 0;
466  //Feedback for completed rotation
467  m_iccComms_o.addTxMsg(ICC_CMD_FB_ROT, 0);
468  //Checks if previous state was a drive state
469  }else if(curState == 1 || curState == 2){
470  // Sets distance parameter of stateflow to zero
471  rtObj.rtU.dist = 0;
472  DPRINT("Distance reached!");
473  //Feedback function for completed distance
474  m_iccComms_o.addTxMsg(ICC_CMD_FB_DIST, 0);
475  }
476  // Checks if a direction change should be performed
477  if(changeDirection){
478  // Starts counter
479  k++;
480  if(k == 50){
481  // Set direction change to false
482  changeDirection = false;
483  // Call API function with reminded value
484  setDistance(remindValue);
485  // Reset reminded value and counter
486  remindValue = 0;
487  k = 0;
488  }
489  }
490  curState = 3;
491  // Reset summed up peaks
492  peak_sum_l = peak_sum_r = 0;
493  m_enc1_o.reset();
494  m_enc2_o.reset();
495  break;
496  case 4:
497  // Case Left Turn state
498  // Set direction pins of H-bridge to a left turn
499  digitalWrite(PIN_DIRECTION_L, LOW);
500  digitalWrite(PIN_DIRECTION_R, LOW);
501  // Checks if first enter of this state is true and the speed is not set via the API function
502  if(curState != 4 && !speed_set){
503  // Reset PI controller
504  myPID.clear();
505  // Reconfigure PI controller
506  myPID.configure(KP_VALUE, KI_VALUE, KD_VALUE, HZ_VALUE, OUTPUT_BITS, OUTPUT_SIGNED);
507  // Set setpoint of the PI controller
508  setpoint_l = spd;
509  setpoint_r = spd;
510  }
511  curState = 4;
512  break;
513  case 5:
514  // Case Right Turn state
515  // Set direction pins of H-bridge to a right turn
516  digitalWrite(PIN_DIRECTION_L, HIGH);
517  digitalWrite(PIN_DIRECTION_R, HIGH);
518  // Checks if first enter of this state is true and the speed is not set via the API function
519  if(curState != 5 && !speed_set){
520  // Reset PI controller
521  myPID.clear();
522  // Reconfigure PI controller

524  // Set setpoint of the PI controller
