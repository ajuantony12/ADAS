105 void CNavigation::getNextState(bool runActive)
106 {
107  if (runActive)
108  {
109  // running navigation run
110  switch (current_state) {
111 
112  case STATE_IDLE:
113  next_state = STATE_GET_OFFSET;
114  cornerMode = false;
115  break;
116 
117  case STATE_GET_OFFSET:
118  cornerMode = false;
119  // Decide if offset is okay or has to be corrected
120  if ((cur_offset < NAV_SET_OFFSET + NAV_TOL_OFFSET) && (cur_offset > NAV_SET_OFFSET - NAV_TOL_OFFSET))
121  { // Offset is in tolerance
122  next_state = STATE_GET_ANGLE;
123  } else {
124  // Offset has to be corrected
125  next_state = STATE_ROT_WALL_INFRONT;
126  }
127  break;
128 
129 
130  case STATE_ROT_WALL_INFRONT:
131  cornerMode = true;
132  if (rotationDone)
133  {
134  next_state = STATE_COR_OFFSET;
135  rotationDone = false;
136  }
137  break;
138 
139  case STATE_COR_OFFSET:
140  cornerMode = true;
141  if (distanceDone)
142  {
143  next_state = STATE_ROT_WALL_OFFSET;
144  distanceDone = false;
145  }
146  break;
147 
148 
149  case STATE_ROT_WALL_OFFSET:
150  cornerMode = true;
151  if (rotationDone)
152  {
153  next_state = STATE_DRIVE_WALL;
154  rotationDone = false;
155  }
156  break;
157 
158 
159  case STATE_GET_ANGLE:
160  if (rotationDone)
161  {
162  cornerMode = false;
163  next_state = STATE_DRIVE_WALL;
164  rotationDone = false;
165  }
166  break;
167 
168 
169  case STATE_DRIVE_WALL:
170  cornerMode = false;
171  if (abs(cur_angle) > NAV_TOL_ANGLE_DRIVE)
172  {
173  next_state = STATE_GET_ANGLE;
174  }
175  break;
176 
177 
178  default:
179  cornerMode = false;
180  next_state = STATE_IDLE;
181  break;
182  }
183  } else {
184  // paused navigation run
185 
186  }
187 }

340 void CNavigation::doTransistionAction(NAV_STATE state, NAV_STATE next)
341 {
342  switch (state)

345  break;
346 
347

352  m_ICC.addTxMsg(ICC_CMD_ROT_ANGLE, -cur_angle);
353 
354  } else if (next == STATE_ROT_WALL_INFRONT)
355  {
356  DPRINTLN2("Nav: Offset has to be corrected...");
357  DPRINT2("Nav: Rotate to wall by "); DPRINT2(-cur_angle + 90); DPRINTLN2(" deg");
358  m_ICC.addTxMsg(ICC_CMD_ROT_ANGLE, -cur_angle + 90);
359  }
360  break;
361 
362 
363  case STATE_ROT_WALL_INFRONT:
364  if (next == STATE_COR_OFFSET)
365  {
366  DPRINTLN2("Nav: Rotation done. Correct offset...");

368  m_ICC.addTxMsg(ICC_CMD_DRIVE_DIST, static_cast<sint16_t>(cur_nxt_wall) - static_cast<sint16_t>(NAV_SET_OFFSET));
369  }
370  break;
371 
372 
373  case STATE_COR_OFFSET:
